pub async fn insert_game(
    db: &Database,
    name: &str,
    mod_dir: &str,
    staging_dir: &str,
    mod_structure: &str,
    requires_elevation: bool,
) -> Result<GameRecord, AppError> {
    let elev = if requires_elevation { 1i64 } else { 0i64 };
    db.execute(
        "INSERT INTO games (name, mod_dir, staging_dir, mod_structure, requires_elevation) VALUES ($1, $2, $3, $4, $5)",
        vec![
            name.into(),
            mod_dir.into(),
            staging_dir.into(),
            mod_structure.into(),
            elev.into(),
        ],
    )
    .await
    .map_err(|e| AppError::DatabaseError(e.to_string()))?;

    // Retrieve the inserted row
    let rows: Vec<(i64, String, String, String, String, i64)> = db
        .select(
            "SELECT id, name, mod_dir, staging_dir, mod_structure, requires_elevation FROM games WHERE rowid = last_insert_rowid()",
            vec![],
        )
        .await
        .map_err(|e| AppError::DatabaseError(e.to_string()))?;

    rows.into_iter()
        .next()
        .map(|(id, name, mod_dir, staging_dir, mod_structure, elev)| GameRecord {
            id,
            name,
            mod_dir,
            staging_dir,
            mod_structure,
            requires_elevation: elev != 0,
        })
        .ok_or_else(|| AppError::DatabaseError("Insert succeeded but row not found".into()))
}

            vec![],
        )
        .await
        .map_err(|e| AppError::DatabaseError(e.to_string()))?;

    Ok(rows
        .into_iter()
        .map(|(id, name, mod_dir, staging_dir, mod_structure, elev)| GameRecord {
            id,
            name,
            mod_dir,
            staging_dir,
            mod_structure,
            requires_elevation: elev != 0,
        })
        .collect())
}

pub async fn update_game(
    db: &Database,
    id: i64,
    name: &str,
    mod_dir: &str,
    staging_dir: &str,
    mod_structure: &str,
    requires_elevation: bool,
) -> Result<GameRecord, AppError> {
    let elev = if requires_elevation { 1i64 } else { 0i64 };
    let rows_affected = db
        .execute(
            "UPDATE games SET name=$1, mod_dir=$2, staging_dir=$3, mod_structure=$4, requires_elevation=$5 WHERE id=$6",
            vec![
                name.into(),
                mod_dir.into(),
                staging_dir.into(),
                mod_structure.into(),
                elev.into(),
                id.into(),
            ],
        )
        .await
        .map_err(|e| AppError::DatabaseError(e.to_string()))?;

    if rows_affected == 0 {
        return Err(AppError::GameNotFound(format!("Game ID {} not found", id)));
    }

    Ok(GameRecord {
        id,
        name: name.to_string(),
        mod_dir: mod_dir.to_string(),
        staging_dir: staging_dir.to_string(),
        mod_structure: mod_structure.to_string(),
        requires_elevation,
    })
}

pub async fn delete_game(db: &Database, id: i64) -> Result<(), AppError> {
    let rows_affected = db
        .execute("DELETE FROM games WHERE id=$1", vec![id.into()])
        .await
        .map_err(|e| AppError::DatabaseError(e.to_string()))?;

    if rows_affected == 0 {
        return Err(AppError::GameNotFound(format!("Game ID {} not found", id)));
    }
    Ok(())
}

use std::path::Path;
use tauri_plugin_sql::Database;

use crate::db::queries::{self, GameRecord};
use crate::error::AppError;
use crate::services::file_ops;

/// Return type for add_game.
/// - cross_drive_warning: staging folder is on a different drive than mod_dir (slower moves)
/// - has_existing_mods: mod_dir already contains .pak/.ucas/.utoc files the user may want to import
#[derive(Debug, serde::Serialize, specta::Type)]
pub struct AddGameResult {
    pub game: GameRecord,
    pub cross_drive_warning: bool,
    pub has_existing_mods: bool,
}

#[tauri::command]
#[specta::specta]
pub async fn add_game(
    db: tauri::State<'_, Database>,
    name: String,
    mod_dir: String,
    staging_dir: Option<String>,
    mod_structure: String,
) -> Result<AddGameResult, AppError> {
    // Compute staging dir: use provided override or default ~/.modtoggler/games/[slug]/staging/
    let staging = match staging_dir {
        Some(s) if !s.is_empty() => s,
        _ => {
            let slug = slug_from_name(&name);
            let home = std::env::var("USERPROFILE")
                .or_else(|_| std::env::var("HOME"))
                .unwrap_or_else(|_| ".".to_string());
            format!("{}/.modtoggler/games/{}/staging", home, slug)
        }
    };

    // Detect if mod_dir is in a protected path (Program Files)
    let requires_elevation = is_protected_path(&mod_dir);

    // Detect cross-drive staging
    let cross_drive = !file_ops::same_volume(Path::new(&mod_dir), Path::new(&staging));

    // Scan mod_dir for existing .pak/.ucas/.utoc files the user might want to import.
    // Best-effort: if mod_dir doesn't exist yet, has_existing_mods = false.
    let has_existing_mods = scan_existing_mods(Path::new(&mod_dir)).await;

    // Create the staging directory on disk (idempotent)
    file_ops::create_staging_dir(Path::new(&staging)).await?;

    let game = queries::insert_game(
        &db,
        &name,
        &mod_dir,
        &staging,
        &mod_structure,
        requires_elevation,
    )
    .await?;

    Ok(AddGameResult {
        game,
        cross_drive_warning: cross_drive,
        has_existing_mods,
    })
}

/// Scan a directory (non-recursively) for .pak, .ucas, or .utoc files.
/// Returns true if any are found, false if the directory is absent, empty, or has none.
async fn scan_existing_mods(mod_dir: &Path) -> bool {
    let mut entries = match tokio::fs::read_dir(mod_dir).await {
        Ok(e) => e,
        Err(_) => return false,
    };
    while let Ok(Some(entry)) = entries.next_entry().await {
        let path = entry.path();
        if let Some(ext) = path.extension().and_then(|e| e.to_str()) {
            let ext_lower = ext.to_ascii_lowercase();
            if matches!(ext_lower.as_str(), "pak" | "ucas" | "utoc") {
                return true;
            }
        }
    }
    false
}

#[tauri::command]
#[specta::specta]
pub async fn remove_game(
    db: tauri::State<'_, Database>,
    id: i64,
) -> Result<(), AppError> {
    // Read game record first to get staging dir path before deleting
    let games = queries::list_games_db(&db).await?;
    let game = games
        .into_iter()
        .find(|g| g.id == id)
        .ok_or_else(|| AppError::GameNotFound(format!("Game ID {} not found", id)))?;

    // Delete from DB (cascades to mods and file_entries)
    queries::delete_game(&db, id).await?;

    // Remove staging directory — best effort, don't fail if already gone
    let staging_path = Path::new(&game.staging_dir);
    if staging_path.exists() {
        let _ = tokio::fs::remove_dir_all(staging_path).await;
    }

    Ok(())
}

/// Convert a game name to a filesystem-safe slug for staging dir path.
fn slug_from_name(name: &str) -> String {
    name.chars()
        .map(|c| if c.is_alphanumeric() || c == '-' { c.to_ascii_lowercase() } else { '-' })
        .collect::<String>()
        .trim_matches('-')
        .to_string()
}

/// Check if a path is inside a Windows protected directory (Program Files).
fn is_protected_path(path: &str) -> bool {
    let lower = path.to_lowercase();
    lower.contains("program files") || lower.contains("program files (x86)")
}

use std::path::Path;
use tauri_plugin_sql::Database;

use crate::db::queries::{self, IntegrityScanResult};
use crate::error::AppError;

/// Run the startup integrity scan.
///
/// Checks:
/// 1. Any in-progress toggle journal entries (crash recovery needed)
/// 2. Enabled mods whose files are missing from the game directory
/// 3. Disabled mods whose files are missing from staging
///
/// IMPORTANT: Returns empty results — not an error — when no mods exist.
/// This is the intentional behavior on first launch (PITFALL-5 guard).
#[tauri::command]
#[specta::specta]
pub async fn run_integrity_scan(
    db: tauri::State<'_, Database>,
) -> Result<IntegrityScanResult, AppError> {
    let mods = queries::list_all_mods(&db).await?;
    let incomplete_journals = queries::scan_incomplete_journals(&db).await?;

    let mut missing_from_game = vec![];
    let mut missing_from_staging = vec![];

    for mod_rec in &mods {
        // Get the game's mod_dir for this mod
        // Note: We need to join through the game — for Phase 1 (no mods), this loop body never runs.
        // In Phase 2+, commands will pass the game mod_dir through context or a join query.
        // For now, read the game record to get mod_dir.
        let file_entries = queries::list_file_entries(&db, mod_rec.id).await?;
        if file_entries.is_empty() {
            // No file manifest yet (mod imported without files, or Phase 1 empty state)
            continue;
        }

        if mod_rec.enabled {
            // Files should exist in the game's mod directory.
            // For integrity check, we check relative to staged_path's sibling game path.
            // In Phase 2 this will use the game record's mod_dir — for now, check staging exists.
            for entry in &file_entries {
                let staging_file = Path::new(&mod_rec.staged_path).join(&entry.relative_path);
                // If enabled, file should NOT be in staging
                if staging_file.exists() {
                    missing_from_game.push(mod_rec.clone());
                    break;
                }
            }
        } else {
            // Files should be in staging
            for entry in &file_entries {
                let staging_file = Path::new(&mod_rec.staged_path).join(&entry.relative_path);
                if !staging_file.exists() {
                    missing_from_staging.push(mod_rec.clone());
                    break;
                }
            }
        }
    }

    Ok(IntegrityScanResult {
        missing_from_game,
        missing_from_staging,
        incomplete_journals,
    })
}
