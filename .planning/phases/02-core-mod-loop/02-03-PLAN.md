  <action>
    1. Create `src-tauri/src/commands/mods.rs` with these commands:

    ```rust
    #[derive(Debug, Serialize, Type)]
    pub struct ImportResult {
        pub mod_record: ModRecord,
        pub file_count: usize,
        pub sub_mods: Vec<SubModRecord>,
        pub has_recognized_files: bool, // false = "No recognized mod files found" warning
    }

    #[tauri::command]
    #[specta::specta]
    pub async fn import_mod(
        app: AppHandle,
        pool: tauri::State<'_, SqlitePool>,
        game_id: i64,
        zip_path: String,
        mod_name: String,
    ) -> Result<ImportResult, AppError>
    ```

    **import_mod implementation:**
    1. Get game record (need staging_dir)
    2. Create mod-specific staging subdir: `{game.staging_dir}/{slug}/` where slug is mod_name lowercased, spaces replaced with hyphens, non-alphanumeric removed
    3. Call `extract_zip_to_staging` via `tokio::task::spawn_blocking` (sync zip crate)
    4. Call `partition_files` on the manifest
    5. Call `has_recognized_mod_files` on the full manifest
    6. Insert mod record (enabled=false, staged_path=mod subdir)
    7. Insert file entries for main_files (sub_mod_id=None)
    8. For each sub-mod folder: insert sub_mod record, insert file entries with sub_mod_id
    9. Return ImportResult

    Other commands (all follow the standard `#[tauri::command] #[specta::specta]` pattern):

    ```rust
    pub async fn list_mods(pool: State<'_, SqlitePool>, game_id: i64) -> Result<Vec<ModRecord>, AppError>
    pub async fn list_mod_files(pool: State<'_, SqlitePool>, mod_id: i64) -> Result<Vec<FileEntry>, AppError>
    pub async fn list_sub_mods_cmd(pool: State<'_, SqlitePool>, mod_id: i64) -> Result<Vec<SubModRecord>, AppError>
    pub async fn toggle_mod_cmd(app: AppHandle, pool: State<'_, SqlitePool>, mod_id: i64, enable: bool) -> Result<(), AppError>
    pub async fn toggle_sub_mod_cmd(app: AppHandle, pool: State<'_, SqlitePool>, sub_mod_id: i64, enable: bool) -> Result<(), AppError>
    pub async fn delete_mod_cmd(app: AppHandle, pool: State<'_, SqlitePool>, mod_id: i64) -> Result<(), AppError>
    pub async fn check_conflicts_cmd(pool: State<'_, SqlitePool>, mod_id: i64, game_id: i64) -> Result<Vec<ConflictInfo>, AppError>
    ```

    These thin wrappers delegate to the service/query layer.

    1. Import `commands` from `../lib/tauri` (NOT `../bindings` -- the lib/tauri.ts re-exports)
    2. Use the same `unwrap` helper pattern
    3. Create these hooks:

    ```typescript
    // Query: list mods for a game
    export function useMods(gameId: number | null) {
      return useQuery({
        queryKey: ['mods', gameId],
        queryFn: async () => {
          if (!gameId) return []
          return unwrap(await commands.listMods(gameId))
        },
        enabled: gameId !== null,
      })
    }

    // Query: list file entries for a mod
    export function useModFiles(modId: number | null) {
      return useQuery({
        queryKey: ['mod-files', modId],
        queryFn: async () => {
          if (!modId) return []
          return unwrap(await commands.listModFiles(modId))
        },
        enabled: modId !== null,
      })
    }

    // Query: list sub-mods for a mod
    export function useSubMods(modId: number | null) {
      return useQuery({
        queryKey: ['sub-mods', modId],
        queryFn: async () => {
          if (!modId) return []
          return unwrap(await commands.listSubModsCmd(modId))
        },
        enabled: modId !== null,
      })
    }

    // Query: check conflicts for a mod
    export function useCheckConflicts(modId: number | null, gameId: number | null) {
      return useQuery({
        queryKey: ['conflicts', modId, gameId],
        queryFn: async () => {
          if (!modId || !gameId) return []
          return unwrap(await commands.checkConflictsCmd(modId, gameId))
        },
        enabled: modId !== null && gameId !== null,
      })
    }

    // Mutation: import mod
    export function useImportMod() {
      const queryClient = useQueryClient()
      return useMutation({
        mutationFn: async (args: { gameId: number; zipPath: string; modName: string }) => {
          return unwrap(await commands.importMod(args.gameId, args.zipPath, args.modName))
        },
        onSuccess: (data) => {
          queryClient.invalidateQueries({ queryKey: ['mods'] })
          if (!data.has_recognized_files) {
            toast.warning('No recognized mod files (.pak/.ucas/.utoc) found in this archive')
          }
          toast.success(`Imported "${data.mod_record.name}" with ${data.file_count} files`)
        },
        onError: (err: Error) => {
          toast.error(err.message ?? 'Import failed')
        },
      })
    }

    // Mutation: toggle mod
    export function useToggleMod() {
      const queryClient = useQueryClient()
      return useMutation({
        mutationFn: async (args: { modId: number; enable: boolean }) => {
          return unwrap(await commands.toggleModCmd(args.modId, args.enable))
        },
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ['mods'] })
          queryClient.invalidateQueries({ queryKey: ['sub-mods'] })
          queryClient.invalidateQueries({ queryKey: ['conflicts'] })
        },
        onError: (err: Error) => {
          toast.error(err.message ?? 'Toggle failed')
        },
      })
    }

    // Mutation: toggle sub-mod
    export function useToggleSubMod() { ... }

    // Mutation: delete mod
    export function useDeleteMod() { ... }
    ```

    IMPORTANT: Use snake_case field names from bindings.ts (mod_record, file_count, has_recognized_files) -- NOT camelCase. This matches the Phase 1 decision.

    IMPORTANT: Check the actual generated function names in bindings.ts after Task 1's cargo build. tauri-specta converts Rust snake_case to camelCase for TS. Adjust hook calls accordingly.
