import { create } from 'zustand'

interface GameStore {
  games: GameRecord[]
  activeGameId: number | null
  setGames: (games: GameRecord[]) => void
  setActiveGame: (id: number | null) => void
}

  activeGameId: null,
  setGames: (games) => set({ games }),

}))
